function! NumberToggle()
  if(&relativenumber == 1)
    set norelativenumber
    set nonumber
  elseif(&number == 1)
    set relativenumber
  else
    set number
  endif
endfunc

nnoremap <C-n> :call NumberToggle()<CR>
